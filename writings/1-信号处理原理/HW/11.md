$$
\begin{align}
X(z)
&= \frac{2z^{-1}-z^{-2}}{1-1.6z^{-1}-0.2z^{-2}} \\
&= \frac{2z^{-1}-z^{-2}}{(1-2z^{-1})(1+0.4z^{-1})} \\
&= A_0+\frac{A_1}{1-2z^{-1}}+\frac{A_2}{1+0.4z^{-1}} \\
\end{align}
$$

$$
\begin{align}
A_0 &= X(0) = \frac{5}{4} \\
A_1 &= \left.(1-2z^{-1})X(z)\right|_{z=2}=\frac{5}{8} \\
A_2 &= \left.(1+0.4z^{-1})X(z)\right|_{z=-0.4}=-\frac{15}{8} 
\end{align} \\
X(z)=1.25+\frac{0.625}{1-2z^{-1}}-\frac{1.875}{1+0.4z^{-1}}
$$
